    class Promise {
    //构造函数
    constructor(executor) {
      this.PromiseState = 'Pending'
  this.PromiseResult = null
  //编写异步回调,解决异步回调
  //this,callback = {}
  //解决多个回调
  this.callbacks = []
  //保存this
  const self = this
  function resolve(data) {
      // console.log(this)
      //1.修改对象的状态,状态只能改变一次
      //判断状态
      if (self.PromiseState !== 'Pending') return;
      self.PromiseState = 'fulfilled'
      //2.修改结果值
      self.PromiseResult = data
      //判断回调函数中是否有函数,解决异步回调的问题
      /*  if(self.callback.onresolved){
           self.callback.onresolved(data)
       } */
      //要保持异步
      setTimeout(() => {
          self.callbacks.forEach(item => {
              item.onresolved(data)
          });
      })

    }
    //reject函数
    function reject(data) {
        //1.修改对象的状态
        //判断状态
        if (self.PromiseState !== 'Pending') return;
        self.PromiseState = 'rejected'
        //2.修改结果值
        self.PromiseResult = data
        //判断回调函数中是否有函数,解决异步
        /*if(self.callback.onrejected){
            self.callback.onrejected(data)
        } */
        //保持异步
        setTimeout(() => {
            self.callbacks.forEach(item => {
                item.onrejected(data)
            })
        })
            ;
    }
    //利用try catch捕获异常
    try {
        executor(resolve, reject)
    } catch (error) {
        reject(error)
    }
}
//then方法
then(onresolved, onrejected) {
    //解决then方法返回的结果是promise对象
    const self = this
    //判断onrejected
    //如果未传值，就自己指定
    if (typeof onrejected !== 'function') {
        onrejected = reason => {
            throw reason;
        }
    }
    if (typeof onresolved !== 'function') {
        onresolved = value => value //es6简写
    }
    return new Promise((resolve, reject) => {
        function callback(type) {
            try {
                //获取回调函数的执行结果    
                let result = type(self.PromiseResult)
                //判断
                if (result instanceof Promise) {
                    //如果是promise对象
                    result.then(v => {
                        resolve(v)
                    }, r => {
                        reject(r)
                    })
                } else {
                    //结果的对象状态为成功
                    resolve(result)
                }
            } catch (error) {
                reject(error)
            }
        }
        //这里不需要try catch也可以
        if (this.PromiseState === 'fulfilled') {
            setTimeout(() => {
                callback(onresolved)
            })

        }
        //指定条件,调用回调函数

        if (this.PromiseState === 'rejected') {
            setTimeout(() => {
                callback(onrejected)
            })


        }
        //因为一旦有异步任务进来,最开始调用then，而此时then里面并不能执行resolve和reject
        if (this.PromiseState === 'Pending') {
            //解决异步回调
            //解决多个回调问题
            this.callbacks.push({
                onresolved: function () {
                    //执行成功回调函数
                    callback(onresolved)

                },
                onrejected: function () {
                    //执行成功回调函数
                    callback(onrejected)

                },
            })
        }
    })
}
//catch方法
catch(onrejected) {
    return this.then(undefined, onrejected)
}
//静态成员，不属于实例对象
static resolve(value) {
    return new Promise((resolve, reject) => {
        if (value instanceof Promise) {
            value.then(v => {
                resolve(v)
            }, r => {
                reject(r)
            })
        } else {
            resolve(value)
        }
    })
}
static reject(reason) {
    return new Promise((resolve, reject) => {
        reject(reason)
    })
}
static all(promises) {
    return new Promise((resolve, reject) => {
        //声明变量
        let count = 0
        let arr = []
        for (let i = 0; i < promises.length; i++) {
            promises[i].then(v => {
                //每个都成功才调用
                count++
                arr[i] = v
                if (count == promises.length) {
                    resolve(arr)
                }
            }, r => {
                reject(r)
            })
        }

    })
}
static race(promises) {
    return new Promise((resolve, reject) => {
        for (let i = 0; i < promises.length; i++) {
            promises[i].then(v => {
                resolve(v)
            }, r => {
                reject(r)
            })
        }
    })
}
